\begin{frame}{Contexto Geral}
\begin{itemize}
  \item Crescimento das aplicações móveis e aumento da complexidade das interfaces.
  \item Flutter consolidado como framework multiplataforma amplamente adotado.
  \item Aplicações ricas exigem mais CPU, GPU e memória; dispositivos móveis têm recursos limitados.
  \item Resultado: desempenho tornou-se um requisito central no desenvolvimento moderno.
\end{itemize}
\end{frame}

\begin{frame}{Problema Real}
\begin{itemize}
  \item Sintomas observados: travamentos, janks, quedas de FPS e uso excessivo de memória.
  \item Reconstruções desnecessárias de widgets e operações pesadas no \texttt{main thread}.
  \item O Flutter oferece abstração e produtividade, porém reduz a visibilidade sobre recursos subjacentes.
  \item Consequência: dificuldade para localizar e corrigir gargalos de desempenho.
\end{itemize}
\end{frame}

\begin{frame}{Sustentabilidade Digital}
\begin{itemize}
  \item Softwares ineficientes consomem mais energia e aceleram desgaste de dispositivos.
  \item Estimativas apontam contribuição significativa de TI nas emissões globais.
  \item Otimização de apps ajuda a reduzir consumo energético e prolongar vida útil do hardware.
  \item Assim, performance e sustentabilidade são preocupações complementares.
\end{itemize}
\end{frame}

\begin{frame}{Limitações das Ferramentas Atuais}
\begin{tabular}{@{}ll@{}}
\toprule
Ferramenta & Limitações relevantes \\
\midrule
Flutter DevTools   & Complexidade; depende de modos específicos (debug/profile).\\
Perfetto           & Dados em baixo nível; análise técnica e manual.\\
Android Profiler   & Focado em Android; não integrado ao app em execução normal.\\
Xcode Instruments  & Restrito a macOS/iOS; coleta manual e custo de uso.\\
\bottomrule
\end{tabular}

\vspace{4mm}
Principais problemas: ferramentas externas, alto custo cognitivo e ausência de recomendações automáticas.
\end{frame}

\begin{frame}{Proposta do Trabalho}
\begin{itemize}
  \item \textbf{collector\_flutter}: ferramenta embarcada que roda dentro do próprio app.
  \item Coleta métricas em tempo real (FPS, frame timing, memória, rebuilds, rede).
  \item Analisa padrões e gera recomendações automáticas.
  \item Projetada para overhead mínimo e fácil integração.
\end{itemize}
\end{frame}

\begin{frame}{Objetivos do Projeto}
\textbf{Objetivo geral:} Desenvolver um protótipo que monitore, analise e recomende otimizações para apps Flutter.

\bigskip
\textbf{Objetivos específicos:}
\begin{itemize}
  \item Coletar métricas com precisão comparável às ferramentas oficiais.
  \item Overhead < 5\%.
  \item Dashboard claro e direto.
  \item Heurísticas úteis para otimização.
  \item Recurso didático para ensino e pesquisa.
\end{itemize}
\end{frame}

\begin{frame}{Metodologia}
\begin{itemize}
  \item Revisão sobre monitoramento e eficiência energética.
  \item Análise de ferramentas atuais.
  \item Desenvolvimento incremental com testes.
  \item Experimentos controlados em dispositivos reais.
  \item Comparação com Android Profiler e ADB.
\end{itemize}
\end{frame}

\begin{frame}{Arquitetura da Solução}
\begin{itemize}
  \item \textbf{Data:} captura via SchedulerBinding, FrameTiming e vm\_service.
  \item \textbf{Domain:} Analyzer + Recommender.
  \item \textbf{Presentation:} Dashboard embutido.
\end{itemize}
\end{frame}

\begin{frame}{Módulos Principais}
\begin{itemize}
  \item \textbf{Collector}
  \item \textbf{Analyzer}
  \item \textbf{Recommender}
  \item \textbf{Dashboard}
  \item \textbf{Exporter}
\end{itemize}
\end{frame}

\begin{frame}{Procedimentos de Teste}
\begin{itemize}
  \item Cenários de carga: rebuilds, memória, rede.
  \item Dispositivos reais.
  \item Repetição de cenários e exportação em JSON.
  \item Comparação com ferramentas externas.
\end{itemize}
\end{frame}

\begin{frame}{Resultados Obtidos}
\begin{itemize}
  \item Precisão: erro médio < 5\%.
  \item Overhead máximo: 2.7\%.
  \item Redução de janks e aumento de FPS.
  \item Dashboard fluido.
\end{itemize}
\end{frame}
